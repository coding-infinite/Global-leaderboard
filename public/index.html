async function getCountryWithEmoji(){
  const res = await fetch("https://ipapi.co/json/");
  const data = await res.json();

  const emoji = data.country_code
    .toUpperCase()
    .replace(/./g, c =>
      String.fromCodePoint(127397 + c.charCodeAt())
    );

  return data.country_name + " " + emoji;
}
